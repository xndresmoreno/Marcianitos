237    def mover(self, teclas):
238        """
239        Mueve la nave según las teclas presionadas.
240        
241        Args:
242            teclas (list): Estado de las teclas del teclado.
243        """
244        if teclas[pygame.K_LEFT] and self.x > 0:

370    def colisiona(self, enemigo):
371        """
372        Verifica colisión con un enemigo.
373        
374        Args:
375            enemigo (Actor): Enemigo con el que verificar colisión.
376            
377        Returns:
378            bool: True si hay colisión.
379        """
380        if enemigo.imagen:
381            enemigo_rect = enemigo.imagen.get_rect(topleft=(enemigo.x, enemigo.y))
382            bala_rect = pygame.Rect(self.x - self.radio, self.y - self.radio, self.radio * 2, self.radio * 2)
383            return enemigo_rect.colliderect(bala_rect)
384        else: 
385            return (enemigo.x < self.x < enemigo.x + enemigo.ancho and
386                    enemigo.y < self.y < enemigo.y + enemigo.alto)

517class Enemigo(Actor):

527        direccion (int): Dirección del movimiento.
528        prob_disparo (float): Probabilidad de disparar en cada frame.
529    """
530    
531    def __init__(self, x, y, nivel):
532        """
533        Inicializa un enemigo.
534        
535        Args:
536            x (float): Posición horizontal inicial.
537            y (float): Posición vertical inicial.
538            nivel (int): Nivel del enemigo (afecta atributos).
539        """
540        super().__init__(x, y) 
541        
542        if nivel == 1:
543            self.imagen = IMAGENES['MARCIANITO_LVL1']
544            self.vida = 1
545        elif nivel == 2:
546            self.imagen = IMAGENES['MARCIANITO_LVL2']
547            self.vida = 1
548        elif nivel == 3:
549            self.imagen = IMAGENES['MARCIANITO_LVL3']
550            self.vida = 2
551        elif nivel == 4:
552            self.imagen = IMAGENES['MARCIANITO_LVL4']
553            self.vida = 2
554        else: 
555            self.imagen = IMAGENES['MARCIANITO_LVL4']
556            self.vida = 3
557
558        self.ancho = self.imagen.get_width()
559        self.alto = self.imagen.get_height()
560        self.velocidad = 2 + nivel * 0.3
561        self.direccion = 1  
562        
563        prob_calculada = 0.0015 + (nivel * 0.0008)
564        self.prob_disparo = min(prob_calculada, 0.02) 
565
566    def actualizar(self):
567        """Actualiza la posición del enemigo."""
568        self.x += self.velocidad * self.direccion
569        if self.x <= 0 or self.x + self.ancho >= PYGAME_CONFIG['ANCHO']:
570            self.direccion *= -1
571            self.y += 20

573    def disparar(self):
574        """
575        Intenta disparar una bala.
576        
577        Returns:
578            list or None: Lista con balas si dispara, None en caso contrario.
579        """
580        if random.random() < self.prob_disparo:
581            return [BalaEnemigo(self.x + self.ancho // 2, self.y + self.alto)] 
582        return None
